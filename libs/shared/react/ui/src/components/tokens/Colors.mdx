import { Meta, ColorPalette, ColorItem } from '@storybook/blocks';
import { colorTokens } from '../../../constants';

# Colors

<ColorPalette>
  {Object.entries(colorTokens).map(([token, values]) => {

    return <ColorItem title={title} colors={values} />;
  })}
